Destructuring

Destructuring

Array Matching (structural assignment)


//Matching
let [first, second] = [1, 2, 3, 4]; // first=1, second=2

//With rest param
let [first, ...rest] = [1, 2, 3, 4]; // rest=[2,3,4]

//Leaving out some values
let [first, , , fourth] = [1, 2, 3, 4]; // first=1, fourth=4

//swapping
[first, second] = [second, first]; // first=1, second=2

with Objects


let hero = {
    firstname: 'Peter',
    lastname: 'Parker'
};

let {firstname, lastname} = hero;
let {firstname:first, lastname:last} = hero;

console.log(`Spiderman is ${first} ${last}`);
// Spiderman is Peter Parker

Object.is

checks if values are the same value

  • strict equality (===) => without type conversion
  • loose equality (==) => perform type conversion
  • Object.is() => same as strict equality with special handling for NaN, -0 and +0


NaN === NaN; //false
Object.is(NaN, NaN); // true

-0 === +0 // true
Object.is(-0, +0); // false

default parameter

default parameter values

old way


// Before ES6

add = (x, y) => {
	x = x || 11;
	y = y || 12;
	console.log( x + y);
}

add();          // 23
add(5, 6);      // 11
add(5);         // 17
add(null, 6);   // 17
add(0, 42);     // 53

default parameter values


// ES6

add = (x = 11, y = 12) => {      // default values
	console.log( x + y );
}

add();         // 23
add(5, 6);     // 11
add(5);        // 17
add(null, 6);  // 6 (null is interpreted as 0)
add(0, 42);    // 42

Class



// Before ES6

var Car = function(name) {
    this.name = name;
}

Car.prototype.getName() {
    return this.name;
}

var car = Car('Elfriede');
console.log(car.getName()); // Elfriede

Class



// ES6

class Car {
  constructor(name) {
    this.name = name;
  }
  
  getName() {
    return this.name;
  }
}

let car = new Car('Elfriede');
console.log(car.getName()); // Elfriede

Child Class



class SpecialCar extends Car {
  constructor(name) {
    super(name);
    this.specialName = 'Special-' + name;
  }
  
  getSpecialName() {
    return this.specialName;
  }
}

let car = new SpecialCar("Elfriede");
console.log(car.getSpecialName()); // Special-Elfriede
console.log(car.getName());        // Elfriede

Symbol Type

Unique and immutable data type as an identifier for object properties


Symbol('foo') !== Symbol('foo');    // true
const foo = Symbol();
const bar = Symbol();
let obj = {};
obj[foo] = 'foo';
obj[bar] = 'bar';
JSON.stringify(obj);                // {}
Object.keys(obj);                   // []
Object.getOwnPropertyNames(obj);    // []
Object.getOwnPropertySymbols(obj);  // [ foo, bar ]

Iterable

support 'iterable' protocol to allow objects to customize their iteration behaviour


let fibonacci = {
    [Symbol.iterator]() {
        let pre = 0, cur = 1;
        return {
           next() {
               [ pre, cur ] = [ cur, pre + cur ];
               return { done: false, value: cur };
           }
        }
    }
}

for (let n of fibonacci) {
    if (n > 1000) { break; }
    console.log(n)
}

Promise

Promise.all(...promises);


// Promise.all(...promises) returns a Promise

// if one fails, returned Promise is rejected

         console.log(`${values[0]} and ${values[1]});
    })
    .catch((error) => {
         console.log(error);
    });

  • can't handle NaN

let numbers = [1, 2, 3, 4, NaN];

  console.log('Array contains value');

Array.prototype.includes()

new way to check if array includes a specific value

  • can handle NaN

// ES7

if(numbers.includes(2)) {
  console.log('Array contains value');
}
console.log(numbers.includes(NaN));      // true

async/await

  • waits until Promise settles and return its result
  • await keyword can only be used in async functions

function fetchTextByPromise() {
  return new Promise(resolve => {
    setTimeout(() => {
      resolve("es8");
    }, 2000);
  });
}

async function sayHello() {
  const text = await fetchTextByPromise();
  console.log(`Hello, ${text}`);
}

console.log(1);
sayHello();
console.log(2);
// 1
// 2
// Hello, es8 (After about 2 sec)
