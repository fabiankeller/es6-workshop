var

function scoped variable


// ES5

function varTest() {
	var x = 31;
	if (true) {
		var x = 71;     // same variable
		console.log(x);     // 71
	}
	console.log(x);     // 71
}

let

block scoped variable


// ES6

function varTest() {
	var x = 31;
	if (true) {
		let x = 71;     // different variable
		console.log(x);     // 71
	}
	console.log(x);     // 31
}

Destructuring

Destructuring

structural assignment


//Matching
let [first, second] = [1, 2, 3, 4];
console.log(first + " and " + second); // 1 and 2

//With rest param
let [first, ...rest] = [1, 2, 3, 4];
console.log(rest); // 2,3,4

//swapping
[first, second] = [second, first];
console.log(first + " and " + second); // 2 and 1

with Objects


let hero = {
    firstname: 'Peter',
    lastname: 'Parker'
};

let {firstname, lastname} = hero;
let {firstname:first, lastname:last} = hero;

console.log(`Spiderman is ${first} ${last}`);
// Spiderman is Peter Parker

Modules

Old way 


// ES5

function Hello(name) {
    function greeting() {
        console.log( "Hello " + name + "!" );
    }

    // public API
    return {
        greeting: greeting
    };

var me = Hello( "Kyle" );
me.greeting();          // Hello Kyle!

Modules

Importing 


// ES6

import * as awesomeModule from "lib/myModule";
awesomeModule.myAwesomeModuleFunction();

default parameter

default parameter values

old way


// ES5

function add(x, y) {
	x = x || 11;
	y = y || 12;
	console.log( x + y);
}

add();      // 23
add(5, 6);      // 11
add(5);     // 17
add(null, 6);       // 17
add(0, 42);     // 53

default parameter values


// ES6

function add(x = 11, y = 12) {      // default values
	console.log( x + y );
}

add();      // 23
add(5, 6);      // 11
add(5);     // 17
add(null, 6);       // 6 (null is interpreted as 0)
add(0, 42);     // 42

default parameter expressions

lazy evaluated



function foo(x = y + 3, z = bar(x)) {

Spread

spread it out into values

	console.log(x, y, z);
}
foo( …[1, 2, 3] );		// 1 2 3

var a = [2, 3, 4];

Rest

collect the rest

	console.log(x, y, z);
}

arrow functions (lambdas)

Arrow Functions

The ES 5 Version with binding this manually 


var reorgBuilder = {
    name: "Awesome new Reorg",

    createNewReorg: function createNewReorg() {
        return function() {
            console.log(this.name +
                " will be finished in: 5 months!"
            );
        }.bind(this);
    }
};

var newReorg = reorgBuilder.createNewReorg();
newReorg();

Arrow Functions

ES 6 Arrow Function and this binding 


let reorgBuilder = {
    name: "Awesome new Reorg",

    createNewReorg() {
        return () => console.log(this.name +
            " will be finished in: 5 months!");
    }
};

let newReorg = reorgBuilder.createNewReorg();
newReorg();

Class

Class



class Car {
  constructor(name) {
    this.name = name;
  }
  
  getName() {
    return this.name;
  }
}

let car = new Car("Elfriede");
console.log(car.getName()); // Elfriede

Child Class



class SpecialCar extends Car {
  constructor(name) {
    super("normal name");
    this.specialName = name;
  }
  
  getSpecialName() {
    return this.specialName;
  }
}

let car = new SpecialCar("Elfriede");
console.log(car.getSpecialName()); // Elfriede
